// Ratinge göre grup (adet dahil)
SELECT rating, COUNT(*) AS film_sayisi
FROM film
GROUP BY rating
ORDER BY rating;

// replacement_cost'a göre grup (film sayısı > 50)
SELECT replacement_cost, COUNT(*) AS film_sayisi
FROM film
GROUP BY replacement_cost
HAVING COUNT(*) > 50
ORDER BY film_sayisi DESC, replacement_cost;

// customer -> store_id (müşteri sayısı)
SELECT store_id, COUNT(*) AS musteri_sayisi
FROM customer
GROUP BY store_id
ORDER BY store_id;

// city -> country_id'ye göre grup (en çok şehri olan ve şehir sayısı)
SELECT country_id, COUNT(*) AS sehir_sayisi
FROM city
GROUP BY country_id
ORDER BY sehir_sayisi DESC
LIMIT 1;
